여러분에게 주어진 첫번째 과제는 RGB 이미지를 자주차에서 가져오는 것입니다. 오른쪽 코드 편집기에 아래과 같은 코드가 표시될 것입니다. 이제 다음페이지로 이동하여 코드를 실행 해 보겠습니다 

from control import Jajucha #자주차 라이브러리 가져오기
import cv2 #opencv 가져오기

def main_loop():
  qrgb,qdepth = jajucha.camera_init()  #자주차의 카메라를 준비
  while True: 
    rgb = jajucha.image_get(qrgb) #자주차의 카메라에서 이미지 가져오기
    jajucha.image_send(rgb) #자주차에서 컴퓨터로 이미지 보내기
      
if __name__ == "__main__":
  jajucha = Jajucha() #자주차 내장 함수 사용을 위한 인스턴스 생성
  jajucha.control(45,45,50) #자주차의 하드웨어 초기화
  main_loop() #메인 루프
